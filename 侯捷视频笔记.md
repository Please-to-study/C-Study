
相同 class 的各个 objects 互为友元

```C++
class complex 
{
public:
  complex(double r = 0, double i = 0): re(r), im(i) {}
  int func(const complex& param)
  {
    return param.re + param.id;
  }
private:
  double re, im;
};
{
  complex c1(2,1);
  complex c2;
  c2.func(c1);
}
```
